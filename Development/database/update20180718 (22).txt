CREATE OR REPLACE FUNCTION wash.get_discount_value_by_key(
  in_key character varying
)
RETURNS integer AS
$BODY$
DECLARE
BEGIN
  RETURN(
    SELECT setting_value
    FROM wash.setting
    WHERE setting_key ilike in_key
  );
END;
$BODY$
  LANGUAGE plpgsql;
